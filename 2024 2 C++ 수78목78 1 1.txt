void save(const std::string& inFileName)
{
	// 전달된 inFileName을 읽기 모드로 연다
	std::ifstream in{ inFileName };


	//저장할 파일을 쓰기 모드로 연다
	std::ofstream out{ "2024 2 C++ 수78목78 1 1.txt", std::ios::app };

	//읽을 파일에서 한 글자를 읽어 쓸 파일에 쓴다.
	char c;
	in >> std::noskipws;
	while (in >> c)
		out << c;

}
