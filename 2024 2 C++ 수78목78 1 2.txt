#include <iostream>
#include <print>
#include <string>
#include <fstream>
#include <chrono>

void save(const std::string&);

int main()
{
	//std::cout << "안녕? C++!" << '\n';
	//std::println({:^80}, "안녕?C++");

	// 한학기 강의를 자동저장하는 함수를 만든다

	save(std::string{ "소스.cpp" });
}

void save(const std::string& inFileName)
{
	// 전달된 inFileName을 읽기 모드로 연다 ->> 별로 좋지 않은 형식 
	std::ifstream in{ inFileName }; // {} - uniform initializer 


	//저장할 파일을 쓰기 모드로 연다
	std::ofstream out{ "2024 2 C++ 수78목78 1 2.txt", std::ios::app };

	//저장한 시간을 기록한다. 
	auto now = std::chrono::system_clock::now();


	out << '\n' << '\n';
	out << "=====================================================" << std::endl;
	out << "저장시간:" << now;
	out << "=====================================================" << std::endl;
	out << '\n';

	//읽을 파일에서 한 글자를 읽어 쓸 파일에 쓴다.
	char c;
	in >> std::noskipws;
	while (in >> c)
		out << c;

}
